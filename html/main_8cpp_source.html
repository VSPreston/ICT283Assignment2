194float CalculateMean(const Vector<float>& array, int size) {
195 float sum =0;
196 for (int i = 0;i< size ; i++) {
197 // add to sum
198 sum += array[i];
199 }
200 sum = sum / size;
201 return sum;
202}

204float CalculateSD(const Vector<float>& array, int size) {
205
206 float mean = CalculateMean(array, size);
207 float sd = 0.0;
208
209 for (int i = 0; i < size; i++) {
210 sd += pow((array[i] - mean),2);
211 }
212
213 return sqrt (sd/(size-1));
214}
