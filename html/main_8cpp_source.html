250float CalculateSD(const Vector<float>& array) {
251
252 float mean = CalculateMean(array);
253 float sd = 0.0;
254
255 for (int i = 0; i < array.Size(); i++) {
256 sd += ((array[i] - mean) * (array[i] - mean));
257 }
258
259 sd = sd/ (array.Size()-1);
260
261 return std::sqrt(sd);
262}

383float sPCC(const Vector<float>& vec1, const Vector<float>& vec2) {
384 float meanX = CalculateMean(vec1);
385 float meanY = CalculateMean(vec2);
386 // std::cout << meanX <<std::endl;
387 // std::cout << meanY <<std::endl;
388
389 // Calculate standard deviations
390 float sdX = CalculateSD(vec1);
391 float sdY = CalculateSD(vec2);
392
393 // std::cout << sdX <<std::endl;
394 // std::cout << sdY <<std::endl;
395
396 float cov = 0;
397 for (int i = 0;i < vec1.Size(); ++i) {
398 cov += (vec1[i] - meanX) * (vec2[i] - meanY);
399 }
400 cov /= vec1.Size();
401
402 float correlation = cov / (sdX * sdY);
403 return correlation;
404}
